## 染色数

$O(n2^n)$

```cpp

#define W 23
#define N 1<<23
typedef bitset<W> bs;
typedef long long ll;
bs g[W];
int w[N]; ll a[N], b[N];

int chromatic_number(int n) {
    fill_n(a, 1 << n, 1); fill_n(b, 1 << n, 1);
    for (int i = 0; i != (1 << n); ++i)
        w[i] = w[i >> 1] + (i & 1);
    for (int i = 0; i != (1 << n); ++i)
        for (int j = 0; j != n; ++j)
            if ((i & (1 << j)) && (g[j] & bs(i)).any())
                a[i] = 0;
    a[0] = 0;
    for (int i = 0; i != n; ++i)
        for (int j = 0; j != (1 << n); ++j)
            if (j & (1 << i)) a[j] += a[j ^ (1 << i)];
    int ans = 0;
    for (int k = 1; !ans; ++k) {
        for (int i = 0; i != (1 << n); ++i) b[i] *= a[i];
        ll s = 0;
        for (int i = 0; i != (1 << n); ++i)
            s += (w[i] & 1) ? -b[i] : b[i];
        if (s) ans = k;
    }
    return ans;
}
```
