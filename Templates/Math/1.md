$$
f(n)=\left\{
\begin{aligned}
1,& n \mod 4 \equiv 1\\
-1,& n \mod 4 \equiv 3\\
0,& n \mod 2 \equiv 0
\end{aligned}\right.
$$

$$
F(n) = 4 \sum_{d | n}f(d)
$$

$$
f(p)=\left\{
\begin{aligned}
p-1,& p \mod 4 \equiv 1\\
p+1,& p \mod 4 \equiv 3\\
0,& other
\end{aligned}\right.
$$

$$
f(p^k)=p^{k-1}f(p)
$$

$$
F(n)=\sum_{d|n}f(d) 
$$
